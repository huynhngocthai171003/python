GUI - ListBox

Đối tượng ListBox hiển thị một danh sách các mục mà từ đó có thể chọn một hoặc nhiều mục.

___init__( self, master, items=None, selected=None, command=None, grid=None, align=None, visible=True, enabled=None, multiselect=False, scrollbar=False, width=None, height=None)

Làm thế nào để tạo một cái? Tạo một đối tượng ListBox như sau:

Ví dụ:
from guizero import App, ListBox app = App() listbox = ListBox(app, items=["Thịt bò", "Thịt gà", "Cá", "Chay"]) app.display()

Các tham số khởi tạo Khi bạn tạo một đối tượng ListBox, bạn phải chỉ định master và có thể chỉ định bất kỳ tham số tùy chọn nào. Chỉ định các tham số trong dấu ngoặc, như sau: listbox = ListBox(app, items=["Thịt bò", "Thịt gà", "Cá", "Chay"])

Nếu bạn muốn ListBox cho phép chọn nhiều mục, bạn phải đặt tham số tùy chọn multiselect thành True: listbox = ListBox(app, multiselect=True)

from guizero import App, ListBox, Text # Hàm để hiển thị các mục đã chọn def show_selected_items(): selected_items = listbox.value # Lấy các mục đã chọn selected_text.value = f"Bạn đã chọn: {', '.join(selected_items)}" # Cập nhật văn bản hiển thị # Tạo ứng dụng app = App(title="Ví dụ ListBox đa chọn") # Tạo một ListBox cho phép chọn nhiều mục listbox = ListBox( app, items=["Thịt bò", "Thịt gà", "Cá", "Chay", "Tôm", "Mực"], multiselect=True, # Cho phép chọn nhiều mục command=show_selected_items # Gọi hàm khi có sự thay đổi ) # Tạo một Text để hiển thị các mục đã chọn selected_text = Text(app, text="Bạn chưa chọn mục nào", size=14) # Hiển thị ứng dụng app.display()
Tham số Nhận vào Mặc định Bắt buộc Mô tả Ví dụ
master App, Window hoặc Box Không Có Bộ chứa mà widget này thuộc về. ListBox(app)
selected chuỗi hoặc Danh sách None Không Mục hoặc các mục được chọn mặc định khi ListBox hiển thị. ListBox(app, selected="Mục 1")
items Danh sách Không Có Danh sách các mục được hiển thị trong ListBox. ListBox(app, items=["Mục 1", "Mục 2", "Mục 3"])
align chuỗi None Không Căn chỉnh widget này trong bộ chứa của nó. Các giá trị hợp lệ: "top", "bottom", "left", "right". ListBox(app, align="left")
command tên hàm None Không Tên hàm được gọi khi một tùy chọn khác được chọn. Hàm phải nhận một hoặc không có tham số. ListBox(app, command=my_function)
grid Danh sách [int, int] None Không Tọa độ [x, y] của widget này trong lưới. Chỉ cần thiết nếu bố cục của master là lưới. ListBox(app, grid=[1,2])
visible boolean True Không Xác định widget có hiển thị hay không. ListBox(app, visible=False)
enabled boolean None Không Xác định widget có thể được kích hoạt hay không. Nếu None, kế thừa từ master. ListBox(app, enabled=True)
multiselect boolean False Không Xác định liệu có cho phép chọn nhiều mục hay không. ListBox(app, multiselect=True)
scrollbar boolean False Không Thêm thanh cuộn dọc nếu có nhiều mục hơn không gian hiển thị. ListBox(app, scrollbar=True)
width kích thước None Không Chiều rộng của widget theo pixel hoặc "fill". ListBox(app, width=200)
height kích thước None Không Chiều cao của widget theo pixel hoặc "fill". ListBox(app, height=150)

Các phương thức Bạn có thể gọi các phương thức sau trên đối tượng ListBox của bạn.

Phương thức Nhận vào Trả về Mô tả Ví dụ
after time (int), command (tên hàm), args (danh sách tham số) Không Lên lịch một lần gọi đến command sau time mili giây. ListBox.after(1000, my_function)
append item (chuỗi) Không Thêm một mục mới vào cuối ListBox. ListBox.append("Mục mới")
cancel command (tên hàm) Không Hủy một cuộc gọi đã lên lịch đến command. ListBox.cancel(my_function)
clear Không Không Xóa tất cả các mục trong ListBox. ListBox.clear()
destroy Không Không Hủy widget. ListBox.destroy()
disable Không Không Vô hiệu hóa widget để nó "xám xịt" và không thể tương tác. ListBox.disable()
enable Không Không Kích hoạt widget. ListBox.enable()
focus Không Không Đưa tiêu điểm vào widget. ListBox.focus()
hide Không Không Ẩn widget khỏi tầm nhìn. ListBox.hide()
insert index (int), item (chuỗi) Không Chèn một mục mới vào ListBox tại chỉ số index. ListBox.insert(1, "Mục mới")
remove item (chuỗi) Boolean Xóa mục đầu tiên khỏi ListBox. Trả về True nếu một mục đã được xóa. ListBox.remove("Mục 1")
repeat time (int), command (tên hàm), args (danh sách tham số) Không Lặp lại command mỗi time mili giây. ListBox.repeat(2000, my_function)
resize width (int), height (int) Không Đặt chiều rộng và chiều cao của widget. ListBox.resize(300, 200)
show Không Không Hiển thị widget nếu nó đã bị ẩn trước đó. ListBox.show()
update_command command (tên hàm) Không Cập nhật hàm để gọi khi một tùy chọn khác được chọn. ListBox.update_command(new_function)

Các thuộc tính Bạn có thể thiết lập và lấy các thuộc tính sau:

Phương thức Kiểu dữ liệu Mô tả Ví dụ
align chuỗi Căn chỉnh của widget này trong bộ chứa widget.align = "left"
bg màu sắc Màu nền của widget widget.bg = "red"
children Danh sách Danh sách các widget trong bộ chứa này. [ListBoxWidget, ListBoxScrollbar] widget.children
enabled boolean True nếu widget được kích hoạt widget.enabled = True
font chuỗi Phông chữ của văn bản widget.font = "Arial"
grid Danh sách Tọa độ [x,y] của widget này widget.grid = [1, 2]
height kích thước Đặt chiều cao của widget theo pixel hoặc "fill" widget.height = 150
items Danh sách Trả về danh sách các mục trong ListBox widget.items = ["Mục 1", "Mục 2"]
master App hoặc Box Bộ chứa mà widget này thuộc về widget.master
value chuỗi Thiết lập hoặc trả về các mục được chọn trong ListBox widget.value = "Mục 1"
visible boolean Nếu widget này có hiển thị widget.visible = False
width kích thước Đặt chiều rộng của widget theo pixel hoặc "fill" widget.width = 200
text_size int Kích thước của văn bản widget.text_size = 14
text_color màu sắc Màu của văn bản widget.text_color = "blue"
text_bold boolean Liệu văn bản có đậm hay không widget.text_bold = True
text_italic boolean Liệu văn bản có nghiêng hay không widget.text_italic = True
text_underline boolean Liệu văn bản có gạch chân hay không widget.text_underline = True
text_overstrike boolean Liệu văn bản có bị gạch chéo hay không widget.text_overstrike = True
tk tkinter.Frame Đối tượng tkinter nội bộ, xem Sử dụng tkinter widget.tk
Tham chiếu đến một thuộc tính dưới dạng .property. Ví dụ, nếu đối tượng ListBox của bạn được gọi là listbox, bạn sẽ viết listbox.value.
Bạn có thể thiết lập thuộc tính (ví dụ listbox.value = "Thịt gà") hoặc lấy giá trị của thuộc tính để sử dụng (ví dụ print(listbox.value)).

Ví dụ Chọn màu văn bản từ một ListBox

Khi một mục trong ListBox được chọn, một hàm sẽ được gọi để thay đổi màu của văn bản.
from guizero import App, ListBox, Text # Hàm để thay đổi màu văn bản def change_color(value): t.text_color = value # Cập nhật màu văn bản theo màu đã chọn # Tạo ứng dụng a = App(title="Chọn màu văn bản") # Tạo một Text để hiển thị t = Text(a, text="Đây là một ListBox", color="black") # Tạo một ListBox với các màu listbox = ListBox( a, items=["red", "green", "blue", "yellow", "purple", "turquoise", "pink", "orange", "black", "brown", "cyan"], selected="black", # Màu mặc định được chọn command=change_color, # Gọi hàm khi có sự thay đổi scrollbar=True # Thêm thanh cuộn ) # Hiển thị ứng dụng a.display()
