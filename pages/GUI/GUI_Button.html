PushButton

Phương thức __init__(self, master, command=None, args=None, text="Button", image=None, pady=10, padx=10, grid=None, align=None, icon=None, visible=True, enabled=None, width=None, height=None)

PushButton hiển thị một nút với văn bản hoặc hình ảnh và sẽ gọi một hàm khi được nhấn.


Cách tạo một PushButton

Tạo một đối tượng PushButton như sau:

from guizero import App, PushButton def do_nothing(): print("Button was pressed") app = App() button = PushButton(app, command=do_nothing) app.display()

Tạo PushButton hình ảnh với ảnh như sau:

from guizero import App, PushButton def do_nothing(): print("A picture button was pressed") app = App() button = PushButton(app, image="button.gif", command=do_nothing) app.display()

Tùy thuộc vào hệ điều hành của bạn và việc bạn đã cài đặt các tính năng hình ảnh bổ sung của guizero sẽ xác định loại hình ảnh bạn có thể sử dụng cho PushButton. Mặc định hỗ trợ GIF và PNG, ngoại trừ macOS chỉ hỗ trợ GIF.


Các tham số khi tạo

Khi bạn tạo một đối tượng PushButton, bạn phải xác định master và có thể xác định bất kỳ tham số tùy chọn nào. Xác định các tham số trong dấu ngoặc, như thế này: button = PushButton(app)

Tham số Giá trị Mặc định Bắt buộc Mô tả Ví dụ Code
master App, Window hoặc Box - Có Container mà widget này thuộc về app = App()
button = PushButton(app)
command Tên hàm None - Tên của hàm được gọi khi nút được nhấn def say_hello():
  print("Hello")
button = PushButton(app, command=say_hello)
align Chuỗi None - Căn chỉnh trong container. Các giá trị: "top", "bottom", "left", "right" button = PushButton(app, align="top")
args Danh sách None - Truyền tham số cho hàm command def show_number(n):
  print(n)
button = PushButton(app, command=show_number, args=[5])
grid List [int, int] None - Tọa độ [x, y] của widget nếu master có layout grid button = PushButton(app, grid=[0, 1])
image Chuỗi None - Đường dẫn hình ảnh. Nếu có image thì text bị ghi đè button = PushButton(app, image="button.gif")
padx int 10 - Đệm ngang giữa văn bản/biểu tượng và cạnh của nút button = PushButton(app, padx=20)
pady int 10 - Đệm dọc giữa văn bản/biểu tượng và cạnh của nút button = PushButton(app, pady=15)
text Chuỗi "Button" - Văn bản hiển thị trên nút button = PushButton(app, text="Click Me")
visible boolean True Không Nếu widget nên hiển thị button.visible = False
enabled boolean None Không Nếu widget nên được kích hoạt. Nếu None thì kế thừa từ master button.enabled = True
width Kích thước None Không Chiều rộng của widget bằng ký tự hoặc pixel nếu là hình ảnh button = PushButton(app, width=100)
height Kích thước None Không Chiều cao của widget bằng ký tự hoặc pixel nếu là hình ảnh button = PushButton(app, height=50)

Các phương thức

Bạn có thể gọi các phương thức sau trên đối tượng PushButton.

Phương thức Giá trị Trả về Mô tả Ví dụ
after thời gian (int), command (tên hàm), args (danh sách tham số) - Lập lịch một lần gọi đến command sau thời gian (ms). (Để gọi lại command, sử dụng repeat()) button.after(1000, show_message)
// Gọi hàm show_message sau 1 giây
cancel command (tên hàm) - Hủy lịch một cuộc gọi đã lên lịch tới command button.cancel(show_message)
// Hủy lịch gọi của show_message
destroy - - Hủy widget button.destroy()
// Xóa nút khỏi giao diện
disable - - Vô hiệu hóa widget để nó bị "mờ" và không thể tương tác button.disable()
// Làm nút mờ đi và không nhấn được
enable - - Kích hoạt widget button.enable()
// Kích hoạt lại nút sau khi disable
focus - - Đặt trọng tâm cho widget button.focus()
// Đặt con trỏ chuột vào nút
hide - - Ẩn widget khỏi tầm nhìn. Phương pháp này sẽ tách widget khỏi bộ quản lý bố cục. button.hide()
// Ẩn nút khỏi giao diện
padding padx (int), pady (int) - Đặt lượng đệm ngang (x) và đệm dọc (y) giữa văn bản/biểu tượng và cạnh của nút. button.padding(20, 10)
// Đệm ngang 20px, đệm dọc 10px
repeat thời gian (int), command (tên hàm), args (danh sách tham số) - Gọi lại command sau khoảng thời gian (ms). Điều này hữu ích để lên lịch một hàm được gọi thường xuyên, ví dụ như cập nhật giá trị đọc từ cảm biến. button.repeat(1000, update_time)
// Cập nhật thời gian mỗi 1 giây
resize chiều rộng (int), chiều cao (int) - Đặt chiều rộng và chiều cao của widget button.resize(100, 50)
// Đặt kích thước nút 100x50 px
show - - Hiển thị widget nếu nó đã bị ẩn trước đó button.show()
// Hiển thị lại nút
toggle - - Thay đổi trạng thái của nút thành trạng thái ngược lại - nếu hiện tại được kích hoạt thì vô hiệu hóa và ngược lại. button.toggle()
// Chuyển đổi trạng thái nút
update_command command (tên hàm), args (Danh sách tùy chọn các tham số sẽ được truyền đến command) - Cập nhật hàm để gọi khi nút được nhấn button.update_command(new_function)
// Thay đổi hàm khi nhấn nút

Các thuộc tính

Bạn có thể đặt và nhận các thuộc tính sau:

Phương thức Kiểu dữ liệu Mô tả Ví dụ
align Chuỗi Căn chỉnh của widget này trong container của nó button.align = "left"
// Căn nút về bên trái trong Box
bg Màu Màu nền của widget button.bg = "red"
// Đặt màu nền của nút thành màu đỏ
enabled boolean True nếu widget được kích hoạt button.enabled = False
// Vô hiệu hóa nút
font Chuỗi Phông chữ của văn bản trên nút button.font = "Arial"
// Đổi phông chữ của nút thành Arial
grid Danh sách Tọa độ [x,y] của widget. Tham số này chỉ được yêu cầu nếu đối tượng master có layout grid button.grid = [0, 1]
// Đặt nút tại dòng 0, cột 1
height Kích thước Đặt chiều cao của widget bằng ký tự hoặc pixel nếu là nút hình ảnh hoặc "fill" button.height = 50
// Đặt chiều cao nút là 50px
image Chuỗi nguồn hình ảnh Đường dẫn tệp, tkinter.PhotoImage hoặc PIL.Image mà bạn muốn hiển thị button.image = "path/to/image.png"
// Thêm hình ảnh vào nút
master App hoặc Box Container mà widget này thuộc về button.master = box
// Đặt Box làm container cho nút
text Chuỗi Văn bản trên nút button.text = "Nhấn vào đây"
// Đổi nội dung nút thành "Nhấn vào đây"
text_color Màu Màu của văn bản trên nút button.text_color = "blue"
// Đổi màu chữ của nút thành xanh dương
text_size int Kích thước của văn bản trên nút button.text_size = 20
// Đặt kích thước chữ là 20px
text_bold boolean Văn bản có đậm không button.text_bold = True
// Làm chữ trên nút đậm hơn
text_italic boolean Văn bản có in nghiêng không button.text_italic = True
// Làm chữ trên nút nghiêng đi
text_underline boolean Văn bản có gạch dưới không button.text_underline = True
// Gạch chân chữ trên nút
text_overstrike boolean Văn bản có gạch ngang không button.text_overstrike = True
// Gạch ngang chữ trên nút
tk tkinter.Button Đối tượng tkinter nội bộ, xem Using tkinter print(button.tk)
// In đối tượng tkinter.Button của nút
value int Trả về 1 khi nút được nhấn, 0 nếu nút được thả ra print(button.value)
// In trạng thái của nút (1 hoặc 0)
visible boolean Nếu widget này hiển thị button.visible = False
// Ẩn nút khỏi giao diện
width Kích thước Đặt chiều rộng của widget bằng ký tự hoặc pixel nếu là nút hình ảnh hoặc "fill" button.width = 100
// Đặt chiều rộng của nút là 100px

Tham chiếu đến thuộc tính như <tên của widget>.property. Ví dụ: nếu đối tượng PushButton của bạn được gọi là button, bạn có thể viết button.value.

Bạn có thể đặt thuộc tính (ví dụ button.bg = "red") hoặc nhận giá trị của thuộc tính để sử dụng (ví dụ print(button.bg)).
